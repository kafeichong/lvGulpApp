//启动配置 
1.git clone git@github.com:kafeichong/lvGulpApp.git 
2.cd lvGulpApp 
3.npm install 
4.gulp  
5.gulp serve 



//命令说明 
```
	1. gulp clean //清除目录; 
	2. gulp serve //启动服务器,并监测编译刷新浏览器; 
	3. gulp scripts //合并压缩js; 
	4. gulp styles //合并压缩scss文件为css; 
	5. gulp lint //js编码规范; 
	6. gulp html //压缩html,并且合并html文件; 
```
